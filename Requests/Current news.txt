<h1>Number Guessing Game</h1>
<p>Guess a number between 1 and 100:</p>
<input type="number" id="guess" min="1" max="100">
<button onclick="checkGuess()">Check</button>
<p id="message"></p>

<script>
  // Generate a random number between 1 and 100
  const randomNumber = Math.floor(Math.random() * 100) + 1;
  let numberOfTries = 0;

  function checkGuess() {
    const guess = parseInt(document.getElementById("guess").value);
    numberOfTries++;

    if (guess === randomNumber) {
      document.getElementById("message").textContent = `Congratulations! You guessed the number ${randomNumber} in ${numberOfTries} tries.`;
      document.getElementById("guess").disabled = true;
    } else if (guess < randomNumber) {
      document.getElementById("message").textContent = "Try a higher number.";
    } else {
      document.getElementById("message").textContent = "Try a lower number.";
    }
  }
</script>
